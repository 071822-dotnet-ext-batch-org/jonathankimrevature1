class Result
{
	public static List<int> gradingStudents(List<int> grades)
	{	
		//HackerRank answer format
		List<int> finalScores = new List<int>();
        
		//Logic to round or not round grades
        	foreach (int n in grades)
        	{
            		if (n >= 38 && n % 5 >= 3)
            		{
                		finalScores.Add(n + (5 - (n % 5)));
           	 	}
           	 	else
           	 	{
                		finalScores.Add(n);
            		}
        	}

		//Output HackerRank solution
        	return finalScores;
	}
}
