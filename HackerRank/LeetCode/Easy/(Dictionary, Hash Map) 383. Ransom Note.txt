public class Solution {
    public bool CanConstruct(string ransomNote, string magazine) {
        Dictionary<char,int> letterPool = new Dictionary<char,int>();
            
        foreach(char letter in magazine){
            if(!letterPool.ContainsKey(letter)){
                letterPool[letter] = 1;
            }
            else{
                letterPool[letter]++;
            }
        }
        
        foreach(char letter in ransomNote){
            if(!letterPool.ContainsKey(letter)){
                return false;
            }
            else if(letterPool[letter] > 0){
                letterPool[letter]--;
            }
            else{
                return false;
            }
        } 
        return true; 
    }
}

/**
1. In Java it's HashMap
   In C# it's Dictionary
2. ".ContainsKey(<key>)" is the important thing here
3. To summarize: Count frequency of each unique word in magazine,
                 If ransomNote does not have a letter, false
                 If ransomNote does have a letter, -1 from that unique word count
                 If ransomeNote does have a letter, but unique word count is 0, false            
**/
