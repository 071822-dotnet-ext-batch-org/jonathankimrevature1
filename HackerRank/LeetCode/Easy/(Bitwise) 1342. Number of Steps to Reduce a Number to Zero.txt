public class Solution {
    public int NumberOfSteps(int num) {
        
        int count = 0;
        int n = num;
        for (int i = 1; i <= num ; i++ )
        {
            bool even = n % 2 == 0;
            bool odd = n %2 != 0;
            bool zero = n == 0;
            
            if (even)
            {
                n = n / 2;
            }
            if (odd)
            {
                n -= 1;
            }
            if (zero)
            {
                break;
            }
            count ++;
        }
        return count ;    
    }
}
