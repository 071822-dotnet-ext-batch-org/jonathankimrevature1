class Result
{
	public static void miniMaxSum(List<int> arr)
	{
		//Declare counters
		int max = 0;
		int min = 0;
		
		//Add up all values in the array arr
		int sum = arr.Sum();
        
		//Logic to calculate max and min
		max = sum - arr.Min();
		min = sum - arr.Max();
        	
		//Output HackerRank solution
		Console.WriteLine(min + " " + max);
	}
}
